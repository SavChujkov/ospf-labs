

# office-gateway настройка запрета обратной редистрибьюции

route-map EIGRP-TO-OSPF deny 10
 match tag 200

route-map EIGRP-TO-OSPF permit 20
 set tag 100

route-map OSPF-TO-EIGRP deny 10
 match tag 100

route-map OSPF-TO-EIGRP permit 20
 set tag 200

router ospf 1
 redistribute eigrp 1 subnets route-map EIGRP-TO-OSPF


#указываем редистрибьюции с фильтром от route-map
router eigrp 1
 redistribute ospf 1 metric 1000 100 255 1 1500 route-map OSPF-TO-EIGRP
 no auto-summary
