


##### настройка разеделения area0 на (area-0 и area2)

# backbone-1 (ABR router)
- для backbone area 0 ненадо удалять!! это ведь ABR
  просто выводим интерфейсы которые дложны будут быть в area 2 из area 0
    router ospf 1
    no network 10.0.10.24 0.0.0.3 area 0
    no network 10.0.10.28 0.0.0.3 area 0

  

# office-main (Internal router)
# office-gateway (ASBR router with EIGRP AS)
# office-backup (ASBR router with EIGRP AS)
- удаляем зону 0 
    no router ospf 1


# backbone-1 (ABR router)
- вводим зону area 2 начиная с ABR area0---area2
    router ospf 2
    auto-cost reference-bandwidth 1000000
    network 10.0.10.0 0.0.0.3 area 2
    network 10.0.10.28 0.0.0.3 area 2

- проверяем что связность между EIGRP-<>-office-backup-<>-backbone-1-<>-DC-Local-Router осталась

# office-main (Internal router)
    router ospf 2
    auto-cost reference-bandwidth 1000000
    network 10.0.10.24 0.0.0.3 area 2
    network 10.0.10.20 0.0.0.3 area 2

# office-gateway (ASBR router with EIGRP AS)
    router ospf 2
    auto-cost reference-bandwidth 1000000
    redistribute eigrp 1 subnets route-map EIGRP-TO-OSPF
    network 10.0.10.20 0.0.0.3 area 2
    network 10.0.10.16 0.0.0.3 area 2

    #делаем редистрибьют из только что включенного ospf 2
    router eigrp 1
    no redistribute ospf 1 metric 1000 100 255 1 1500 route-map OSPF-TO-EIGRP
    redistribute ospf 2 metric 1000 100 255 1 1500 route-map OSPF-TO-EIGRP
    
# office-backup (ASBR router with EIGRP AS)
    router ospf 2
    auto-cost reference-bandwidth 1000000
    redistribute eigrp 1 subnets route-map EIGRP-TO-OSPF
    network 10.0.10.16 0.0.0.3 area 2
    network 10.0.10.28 0.0.0.3 area 2

    no router ospf 1

    #делаем редистрибьют из только что включенного ospf 2
    router eigrp 1
    no redistribute ospf 1 metric 1000 100 255 1 1500 route-map OSPF-TO-EIGRP
    redistribute ospf 2 metric 1000 100 255 1 1500 route-map OSPF-TO-EIGRP
