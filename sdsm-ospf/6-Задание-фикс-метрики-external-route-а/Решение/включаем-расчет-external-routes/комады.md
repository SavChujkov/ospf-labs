
# смотрим внешние маршруты
 - show ip ospf database external
 - show ip ro  #ищем маршрут E2 или E1 и смотрим стоймость

# настройка включения рассчета для 'default information originate'
 - default-information originate metric 10 metric-type 1
 - metric 10 начальная стоймость с которой маршрут придет в протокол
 - metric-type 1 - изменяем тип на E1 для того чтобы рассчет включился для этого внешнего роута
